Specializations = require '../models/specializations'
_ = require 'lodash'
index =(req, res, next)-> 
  Specializations.find {}, (err, specializations)->
    res.json specializations

setup = (app)->
  Specializations.find {}, (err, specializations)->
    unless specializations.length
      defaultSpecializationsList = [
        "Urologia"
        "Onkologia"
        "Neurologia"
        "Urologia NT"
        "Urologia BPH"
        "Kardiologia"
        "Psychiatria"
        "Okulistyka"
        "Reumatologia"
        "Dermatologia"
      ]

      _.each defaultSpecializationsList, (spToSaveName)->
        spToSave = new Specializations name: spToSaveName
        Specializations.create spToSave, (err, doc)->
          console.log 'controllers/Specializations:: init with defaults'



  app.get '/specializations', index
    
module.exports = setup
